FROM python:3.9-slim # Set the working directory in the container
WORKDIR /app # Copy the server code into the container 
COPY server.py . # Install dependencies
RUN pip install fastapi uvicorn websockets # Expose the port that the FastAPI server will run on
EXPOSE 8000 # Run the FastAPI server with Uvicorn 
CMD ["uvicorn", "server:app", "--host", "0.0.0.0", "--port", "8000"]
